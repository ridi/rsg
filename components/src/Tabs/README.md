### Default
```js
const { Tabs, Tab } = require('@ridi/rsg');

<Tabs>
  <Tab url="#" title="일반도서" subtitle="30" active />
  <Tab url="#" title="로맨스" subtitle="512" />
  <Tab url="#" title="판타지 무협" subtitle="12" />
  <Tab url="#" title="만화" subtitle="3" />
</Tabs>
```

const { Tabs, Tab } = require('@ridi/rsg');
